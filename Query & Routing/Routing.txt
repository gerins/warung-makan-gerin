http://localhost:8080/menus/(inputProductID)		(GET)		// View Single menu availalble
http://localhost:8080/menus				            (POST)		// Add new menu
http://localhost:8080/menus/(inputProductID)		(PUT)		// Update availalble menu
http://localhost:8080/menus/(inputProductID)		(DELETE)	// Delete availalble menu

{
     "menuname": "Nasi Kucing",
     "harga": 2000,
     "stock": 22,
     "category": "1"
 }
